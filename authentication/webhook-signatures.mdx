## Why this matters

Webhooks land at your public endpoint over HTTPS. HTTPS protects the contents in transit, but it doesn't prove the sender is Fibo. Without signature verification, anyone who guesses (or learns) your webhook URL could send fake events.

Every webhook Fibo emits carries an `X-Fibo-Signature` header. Your handler **must** verify it before acting on the body. Drop unsigned or invalid requests with 401.

## Signature scheme

We use **HMAC-SHA256** with a per-tenant secret. The secret is shown once when you create the webhook endpoint (or rotate the secret) and is never displayed again — store it in your server environment alongside your `sk_…` key.

The signed payload is the **raw request body** (the JSON bytes — pre-parsing, no whitespace normalization). The header has the shape:

```
X-Fibo-Signature: t=1716212345,v1=4f7a3b2e9c8d1a6f5e4b3a2d1c0b9a8e7f6e5d4c3b2a1908f7e6d5c4b3a29180
```

- `t` — Unix timestamp (seconds) when the signature was computed. Reject anything older than 5 minutes to prevent replay.
- `v1` — Hex-encoded HMAC-SHA256(`{secret}`, `{t}.{body}`).

Constant-time compare the computed signature against `v1`. **Do not use `==`** — timing attacks are within reach of a determined adversary.

## Reference implementation

<Tabs>
  <Tab title="Node.js (Express)">
    ```ts
    import crypto from 'crypto';
    import type { Request, Response, NextFunction } from 'express';

    const SECRET = process.env.FIBO_WEBHOOK_SECRET!;
    const TOLERANCE_SECONDS = 300;

    export function verifyFiboSignature(req: Request, res: Response, next: NextFunction) {
      const header = req.header('x-fibo-signature');
      if (!header) return res.sendStatus(401);

      const parts = Object.fromEntries(
        header.split(',').map((kv) => kv.split('=').map((s) => s.trim()))
      );
      const ts = parseInt(parts.t, 10);
      const sig = parts.v1;
      if (!ts || !sig) return res.sendStatus(401);

      // Replay window
      if (Math.abs(Date.now() / 1000 - ts) > TOLERANCE_SECONDS) {
        return res.sendStatus(401);
      }

      // req.body must be the raw bytes — use express.raw() on this route.
      const expected = crypto
        .createHmac('sha256', SECRET)
        .update(`${ts}.`)
        .update(req.body)
        .digest('hex');

      const a = Buffer.from(expected, 'hex');
      const b = Buffer.from(sig, 'hex');
      if (a.length !== b.length || !crypto.timingSafeEqual(a, b)) {
        return res.sendStatus(401);
      }

      next();
    }
    ```

    Mount with `express.raw({ type: 'application/json' })` on the webhook route only — *not* globally — so the rest of your API keeps its parsed JSON bodies:

    ```ts
    app.post(
      '/fibo/webhooks',
      express.raw({ type: 'application/json' }),
      verifyFiboSignature,
      (req, res) => {
        const event = JSON.parse(req.body.toString('utf8'));
        // ...handle event...
        res.sendStatus(204);
      },
    );
    ```

  <Tab title="Python (FastAPI)">
    ```python
    import hashlib
    import hmac
    import time
    from fastapi import Header, HTTPException, Request

    SECRET = os.environ["FIBO_WEBHOOK_SECRET"].encode()
    TOLERANCE_SECONDS = 300

    async def verify_fibo_signature(
        request: Request,
        x_fibo_signature: str = Header(...),
    ) -> bytes:
        parts = dict(p.strip().split("=", 1) for p in x_fibo_signature.split(","))
        ts = int(parts.get("t", 0))
        sig = parts.get("v1", "")

        if abs(time.time() - ts) > TOLERANCE_SECONDS:
            raise HTTPException(401, "stale_signature")

        body = await request.body()
        expected = hmac.new(
            SECRET, f"{ts}.".encode() + body, hashlib.sha256
        ).hexdigest()

        if not hmac.compare_digest(expected, sig):
            raise HTTPException(401, "invalid_signature")

        return body
    ```

  <Tab title="Go">
    ```go
    import (
        "crypto/hmac"
        "crypto/sha256"
        "encoding/hex"
        "io"
        "net/http"
        "strconv"
        "strings"
        "time"
    )

    var secret = []byte(os.Getenv("FIBO_WEBHOOK_SECRET"))
    const toleranceSeconds = 300

    func verifyFiboSignature(next http.HandlerFunc) http.HandlerFunc {
        return func(w http.ResponseWriter, r *http.Request) {
            header := r.Header.Get("X-Fibo-Signature")
            var ts int64
            var sigHex string
            for _, kv := range strings.Split(header, ",") {
                parts := strings.SplitN(strings.TrimSpace(kv), "=", 2)
                switch parts[0] {
                case "t":
                    ts, _ = strconv.ParseInt(parts[1], 10, 64)
                case "v1":
                    sigHex = parts[1]
                }
            }
            if abs(time.Now().Unix()-ts) > toleranceSeconds {
                http.Error(w, "stale", 401); return
            }
            body, _ := io.ReadAll(r.Body)
            mac := hmac.New(sha256.New, secret)
            mac.Write([]byte(strconv.FormatInt(ts, 10) + "."))
            mac.Write(body)
            if !hmac.Equal(mac.Sum(nil), mustDecodeHex(sigHex)) {
                http.Error(w, "invalid", 401); return
            }
            r.Body = io.NopCloser(bytes.NewReader(body))
            next(w, r)
        }
    }
    ```

## Common mistakes

- **Verifying after JSON parsing.** Frameworks that auto-parse JSON re-serialize when you stringify — and the bytes won't match. Verify against the raw incoming bytes, then parse.
- **Trusting the `Date` header instead of `t`.** Clocks drift. The `t` field in the signature header is the canonical timestamp.
- **Skipping the replay window.** Without a tolerance check, an attacker who captures one valid webhook can replay it indefinitely.
- **String-compare with `==`.** Timing attacks recover bytes one at a time. Use `timingSafeEqual` / `hmac.compare_digest` / `hmac.Equal`.
- **Rotating the secret without atomic deploy.** When you rotate, the dashboard gives you a brief window where both old and new secrets are accepted by Fibo — use it to deploy the new secret to your server first, then revoke the old one.
